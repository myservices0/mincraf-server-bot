async function deployBot() { const ip = document.getElementById('srvIp').value; const port = document.getElementById('srvPort').value; const statusDiv = document.getElementById('status'); if(!ip || !port) { statusDiv.innerText = "Error: Fill out all parameters."; return; } statusDiv.style.color = "#ffeb3b"; statusDiv.innerText = "Waking up cloud server and connecting... (May take 30s)"; try { // Increased timeout loop to handle cloud service spin-ups const response = await fetch('/api/deploy', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ ip, port: parseInt(port) }), signal: AbortSignal.timeout(45000) // Gives the host 45 seconds to wake up }); const data = await response.json(); if(data.success) { statusDiv.style.color = "#00e676"; statusDiv.innerText = "Status: Bot deployed successfully!"; } else { statusDiv.style.color = "#ff4a4a"; statusDiv.innerText = "Server Error: " + data.message; } } catch (err) { statusDiv.style.color = "#ff4a4a"; statusDiv.innerText = "Cloud backend is waking up. Wait 10 seconds and click Deploy again!"; } }
